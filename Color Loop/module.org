    <script>
        let _state;

        /**
         * Spaltenanzahl basierend auf Anzahl Dots UND verfügbarer Breite.
         *
         * Logik:
         *  - Berechne wie viele Dots nebeneinander passen (max 4, min MIN_DOT_PX)
         *  - Nimm so viele Spalten wie möglich (= eine Reihe wenn Platz reicht)
         *  - Nur wenn nicht alle in eine Reihe passen: wähle möglichst
         *    quadratisches Layout mit minimalen Leerstellen
         */
        const MIN_DOT_PX = 32;
        const MAX_COLS = 4;

        function calcCols(n) {
            const gridEl = document.getElementById('color-grid');
            const availW = gridEl ? gridEl.offsetWidth : 200;
            const gap = 8;

            // Wie viele Spalten passen bei MIN_DOT_PX?
            const maxByWidth = Math.max(1, Math.min(
                Math.floor((availW + gap) / (MIN_DOT_PX + gap)),
                MAX_COLS
            ));

            // Wenn alle Dots in eine Reihe passen → eine Reihe
            if (n <= maxByWidth) {
                console.log(`calcCols: n=${n}, availW=${availW}px, maxByWidth=${maxByWidth} → cols=${n} (score=${Infinity.toFixed(2)})`);
                return n;
            }

            // Sonst: finde beste Spaltenanzahl (quadratisch, wenig Reste, mehr Spalten bevorzugt)
            let best = maxByWidth;
            let bestScore = Infinity;
            for (let c = 1; c <= maxByWidth; c++) {
                const rows = Math.ceil(n / c);
                const remainder = (c - (n % c)) % c;
                // Reste stark bestrafen, quadratisch bevorzugen, bei Gleichstand mehr Spalten
                const score = remainder * 6 + Math.abs(c - rows) - c * 0.01;
                if (score < bestScore) { bestScore = score; best = c; }
            }
            console.log(`calcCols: n=${n}, availW=${availW}px, maxByWidth=${maxByWidth} → cols=${best} (score=${bestScore.toFixed(2)})`);
            return best;
        }

        function applyState() {
            if (!_state) return;
            setBtn('btn-active', 'icon-btn', _state.active);
            setBtn('btn-autostart', 'sw-btn', _state.autostart);
            setBtn('btn-resume', 'sw-btn', _state.resume);
            document.getElementById('slider-step').value = _state.step;
            document.getElementById('val-step').textContent = _state.step + '°';
            document.getElementById('slider-speed').value = _state.speed;
            document.getElementById('val-speed').textContent = _state.speed + 's';

            renderDots();
        }

        function setBtn(id, cls, on) {
            document.getElementById(id).classList.toggle('on', on);
        }

        function setColor(entry, hex) {
            entry.color = hex;
            requestAction('color_' + entry.id, hex === '' ? -1 : parseInt(hex.replace('#', ''), 16));
            renderDots();
        }

        function renderDots() {
            const grid = document.getElementById('color-grid');
            grid.innerHTML = '';
            const n = _state.colors.length;
            const cols = calcCols(n);
            grid.style.setProperty('--dot-cols', cols);
            _state.colors.forEach((dev, i) => {
                const wrap = document.createElement('div');
                wrap.className = 'dot-wrap';

                if (dev.color === '') {
                    // Transparenter Dot – Klick öffnet hidden color-input zum Befüllen
                    const inp = document.createElement('input');
                    inp.type = 'color';
                    inp.style.border = '2px dashed var(--content-color)';
                    inp.style.opacity = '0.2';
                    inp.title = dev.name + ' (transparent, neue Auswahl nicht rückgängig machbar)';
                    inp.value = getComputedStyle(document.documentElement).getPropertyValue('--card-color').trim();
                    inp.addEventListener('change', e => setColor(dev, e.target.value));

                    wrap.appendChild(inp);
                } else {
                    // Befüllter Dot: color-input
                    const inp = document.createElement('input');
                    inp.type = 'color';
                    inp.value = dev.color;
                    inp.title = dev.name;
                    inp.addEventListener('change', e => setColor(dev, e.target.value));

                    wrap.appendChild(inp);
                }
                grid.appendChild(wrap);
            });
        }

        ['btn-active', 'btn-autostart', 'btn-resume'].forEach(id => {
            document.getElementById(id).addEventListener('click', () => {
                const key = id.replace('btn-', '');
                _state[key] = !_state[key];
                document.getElementById(id).classList.toggle('on', _state[key]);
                requestAction(key, _state[key]);
            });
        });
        document.getElementById('slider-step').addEventListener('change', e => {
            _state.step = parseInt(e.target.value);
            requestAction('increment', _state.step);
        });
        document.getElementById('slider-speed').addEventListener('change', e => {
            _state.speed = parseInt(e.target.value);
            requestAction('transition', _state.speed);
        });

        // Hier werden die Daten vom Modul verarbeitet.
        function handleMessage(data) {
            const decodedData = JSON.parse(data);
            console.log('Daten:', decodedData);
            _state = decodedData;
            applyState();
        }
